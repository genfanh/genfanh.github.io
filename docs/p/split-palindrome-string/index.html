func partition(s string) [][]string {
ans := make([][]string,0)
	size := len(s)
	if size == 0 {
		return ans
	}
	// é¢„å¤„ç†
	// ä½¿ç”¨ dp[i][j] è¡¨ç¤º s[i:j] æ˜¯å¦å›æ–‡
	dp := make([][]bool, size)
	for i := range dp {
		dp[i] = make([]bool, size)
	}
	for r:=0; r<size; r++ {
		for l:=0; l<=r; l++ {
			if s[l] == s[r] && (r - l <=2 || dp[l+1][r-1]) {
				dp[l][r] = true
			}
		}
	}

	path := make([]string, 0)

	var backtracking func(start int)
	backtracking = func (start int) {
		if start >= size {
			tmp := make([]string, len(path))
			copy(tmp, path)
			ans = append(ans, tmp)
			return
		}
		for i := start; i < size; i++ {
			if !dp[start][i] {
				continue
			}
			path = append(path, s[start:i+1])
			backtracking(i+1)
			path = path[:len(path)-1]
		}
	}
	backtracking(0)
	return ans
}
