func wordBreak(s string, wordDict []string) bool {
		ans := make([]string, 0)
	size := len(s)
	// ‰ΩøÁî® map Âä†ÈÄüÊü•Êâæ
	wordMap := make(map[string]bool, size + 1)
	for _, word := range wordDict {
		wordMap[word] = true
	}

	// Êü•ÊâæÊòØÂê¶ÊúâËß£
	dp := make([]bool, size+1)
	dp[0] = true
	for i:=0; i<size; i++ {
		for j:=i+1; j<=size; j++ {
			if dp[i] && wordMap[s[i:j]] {
				dp[j] = true
			}
		}
	}

	// ÂõûÊ∫ØÁÆóÊ≥ïÔºåËé∑ÂèñÊâÄÊúâËß£
	path := make([]string, 0)
	var backtracking func(start int)
	backtracking = func(start int) {
		if start == size && dp[size] {
			tmp := strings.Join(path, " ")
			ans = append(ans, tmp)
			return
		}
		for i := start+1; i<=size; i++ {
			if dp[start] && wordMap[s[start:i]] {
				path = append(path, s[start:i])
				backtracking(i)
				path = path[:len(path)-1]
			}
		}
	}
	if dp[size] {
		backtracking(0)
	}
	return ans
}
