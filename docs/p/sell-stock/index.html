func maxProfit(k int, prices []int) int {
    pLen := len(prices)
    if len(prices) == 0{
        return 0
    }

    //ä¸€æ¬¡äº¤æ˜“éœ€è¦2å¤©ï¼Œå¦‚æœäº¤æ˜“æ¬¡æ•°å¤§äºæ€»çš„å¤©æ•°/2ï¼Œé‚£ä¹ˆå¯ä»¥è®¤ä¸ºkæ˜¯æ— ç©·å¤§ï¼Œå¥—æ¡†æ¶
    if k > pLen/2{
        return maxProfit_k_Inf(prices)
    }

    tmp := make([][2]int, k+1)
    dp := make([][][2]int, 0)
    for i:=0; i<pLen;i++{
        dp = append(dp, tmp)
    }
  
    for i:=0; i<pLen; i++{
        //base case
        if i == 0 {
            for d:=0; d<=k; d++{
                dp[i][d][0]=0
                dp[i][d][1]=-prices[i]  
            }
            continue
        }

        for j:=k; j>=1; j--{
            dp[i][j][0]=max(dp[i-1][j][0], dp[i-1][j][1]+prices[i])
            dp[i][j][1]=max(dp[i-1][j][1], dp[i-1][j-1][0]-prices[i])
        }
    }
    return dp[pLen-1][k][0]
}
