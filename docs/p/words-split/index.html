func wordBreak(s string, wordDict []string) bool {
	size := len(s)
	m := make(map[string]bool, len(wordDict))
	for _, word := range wordDict {
		m[word] = true
	}
	dp := make([]bool, size+1)
	dp[0] = true
	for i:=0; i<size; i++ {
		for j:=i+1; j<=size; j++ {
			if dp[i] && m[s[i:j]] {
				dp[j] = true
			}
		}
	}
	return dp[size]
}
