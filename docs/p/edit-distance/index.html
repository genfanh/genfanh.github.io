func minDistance(word1 string, word2 string) int {
    n, m := len(word1), len(word2)
    dp := make([][]int, n+1)
    //  word2 ä¸ºç©ºæ—¶ word1 å˜ä¸º word2 æœ€å°‘è¦èµ°çš„æ­¥æ•°
    for i:=0; i<=n; i++ {
        dp[i] = make([]int, m+1)
        dp[i][0] = i
    }
    //  word1 ä¸ºç©ºæ—¶å˜ä¸º word2 æœ€å°‘è¦èµ°çš„æ­¥æ•°
    for j:=0; j<=m; j++ {
        dp[0][j] = j
    }
    for i:=1; i<=n; i++ {
        for j:=1; j<=m;j++ {
            if word1[i-1] == word2[j-1] {
                dp[i][j] = dp[i-1][j-1]
            } else {
                dp[i][j] = 1 + min(dp[i-1][j], min(dp[i][j-1], dp[i-1][j-1]))
            }
        }
    }
    return dp[n][m]
}

func min(x, y int) int {
    if x < y {
        return x
    }
    return y
}
